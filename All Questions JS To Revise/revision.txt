            // ‚úÖ String Property

            // ‚úÖ String Methods (One-liner summaries)

        1) Searching Methods : 

            // .startsWith(substring) ‚Üí Checks if string starts with the given substring
            console.log(string1.startsWith("Hello")); // true

            // .endsWith(substring) ‚Üí Checks if string ends with the given substring
            console.log(string1.endsWith("World!")); // true

            // .includes(substring) ‚Üí Checks if string contains the given substring
            console.log(string1.includes("World")); // true

        2) To get the index of the string or character : 

            // .indexOf(char/substring, fromIndex) ‚Üí Returns index of first match, or -1 if not found
            console.log(string1.indexOf('o', 5)); // Output: 8

            // .lastIndexOf(char/substring, fromIndex) ‚Üí Returns last match index before given position
            console.log(string1.lastIndexOf('o', 7)); // Output: 4

        3) to get the character at the particular index : 

            // .charAt(index) ‚Üí Returns the character at the specified index
            console.log(string1.charAt(2)); // Output: l

        4) to get the substring : 

            // .substring(start, end) ‚Üí Extracts substring from start to end (excluding end)
            console.log(string1.substring(0, 5)); // Output: Hello

            // .slice(start, end) ‚Üí Extracts part of the string (accepts negative indices)
            console.log(string1.slice(0, 5)); // Output: Hello
            console.log(string1.slice(-6)); // Output: World!

        5) to convert the string into Array : 

            // .split(separator) ‚Üí Splits string into an array by given separator
            console.log(string1.split("")); // Output: Each character
            console.log(string1.split(" ")); // Output: Words

        6) To Concatinate the two Strings : 

            // .concat(string) ‚Üí Concatenates the given string(s)
            console.log(string1.concat(string2)); // Output: Hello, World! How are you?

            // + operator ‚Üí Another way to concatenate strings
            console.log(string1 + string2); // Output: Hello, World! How are you?

        7) To Convert the String into the uppercase and lowercase :

            // .toUpperCase() ‚Üí Converts string to uppercase
            console.log(string1.toUpperCase()); // Output: HELLO, WORLD!

            // .toLowerCase() ‚Üí Converts string to lowercase
            console.log(string1.toLowerCase()); // Output: hello, world!

            // .trim() ‚Üí Removes leading and trailing whitespace
            console.log(stringWithSpaces.trim()); // Output: Hello, World!

17) Array important methods and properties :

            // ‚úÖ Array Property
            // .length ‚Üí Returns the number of elements in the array
            console.log(array.length); // Output: total elements count

        1) Methods to Add/Remove Elements (Modify Original Array):

            // .push(element1, element2) ‚Üí Adds elements at the end of the array
            array.push("new1", "new2");

            // .pop() ‚Üí Removes the last element of the array
            array.pop();

            // .unshift(element1, element2) ‚Üí Adds elements at the beginning of the array
            array.unshift("start1", "start2");

            // .shift() ‚Üí Removes the first element of the array
            array.shift();

            // .splice(startIndex, deleteCount, ...items) ‚Üí Add/remove elements at any position
            array.splice(1, 2, "insert1", "insert2");

            // .reverse() ‚Üí Reverses the array elements in-place
            array.reverse();

        2) Iteration Methods (Higher-Order Functions - Do NOT Modify Original Array):

            // .forEach(callback) ‚Üí Iterates over array elements, no return value
            array.forEach((element, index, arr) => {});

            // .map(callback) ‚Üí Returns a new array with modified values
            const mapped = array.map((element, index, arr) => element + 1);

            // .filter(callback) ‚Üí Returns a new array with elements that pass the test
            const filtered = array.filter((element, index, arr) => element > 10);

            // .find(callback) ‚Üí Returns the first element that satisfies the test
            const found = array.find((element, index, arr) => element === 20);

            // .reduce(callback, initialValue) ‚Üí Reduces array to a single value from left to right
            const sum = array.reduce((acc, curr) => acc + curr, 0);

            // .reduceRight(callback, initialValue) ‚Üí Same as reduce but from right to left
            const sumRev = array.reduceRight((acc, curr) => acc + curr, 0);

        3) Methods That Return Boolean:

            // .every(callback) ‚Üí Returns true if all elements pass the test
            const allPass = array.every((element) => element > 0);

            // .some(callback) ‚Üí Returns true if any element passes the test
            const anyPass = array.some((element) => element > 50);

        4) Search and Index Methods (Similar to String):

            // .indexOf(element) ‚Üí Returns first index of the element, or -1 if not found
            const index = array.indexOf(30);

            // .lastIndexOf(element) ‚Üí Returns last index of the element, or -1
            const lastIndex = array.lastIndexOf(30);

        5) Combining or Extracting Elements:

            // .concat(array2) ‚Üí Merges arrays and returns a new array
            const merged = array1.concat(array2);

            // .slice(start, end) ‚Üí Extracts a portion of the array (non-mutating)
            const part = array.slice(1, 3);

        6) Convert Array to String:

            // .toString() ‚Üí Converts the array to a comma-separated string
            const str = array.toString();

            // .join(separator) ‚Üí Joins array into a string with specified separator
            const joined = array.join(" - ");

        7) Sorting:

            // .sort() ‚Üí Sorts the array based on Unicode (can be customized)
            array.sort();

            // .sort((a, b) => a - b) ‚Üí Sorts array in ascending numerical order
            array.sort((a, b) => a - b);

            // .sort((a, b) => b - a) ‚Üí Sorts array in descending numerical order
            array.sort((a, b) => b - a);

        8) Array Creation Methods:

            // Array.from(iterable) ‚Üí Converts iterable (like string) to array
            Array.from("hello"); // ['h','e','l','l','o']

            // Array.of(...elements) ‚Üí Creates an array from arguments
            Array.of(1, 2, 3); // [1, 2, 3]

            // new Array(size) ‚Üí Creates an array of given length
            new Array(5); // [empty √ó 5]

// ‚úÖ 8. Value Copy vs Reference Copy

// üîπ Value Copy (Primitive Types)
let num1 = 10;
let num2 = num1;
num1 = 45;  // num2 remains 10

// üîπ Reference Copy (Non-Primitive - like objects)
let objRef1 = {
    name: "Prithvi",
    car: { model: "Mercedes", price: "1.3Cr" }
};
let objRef2 = objRef1;
objRef1.name = "Anant";  // Affects objRef2 too

//------------------------------------------------------

// ‚úÖ 9. Shallow Copy

let shallowCopy = { ...objRef1 };  // new object, but nested objects share reference

objRef1.car.model = "Tata Harrier";  // Also affects shallowCopy

//------------------------------------------------------

// ‚úÖ 10. Deep Copy (Solves nested mutation issue)

// Use JSON methods for deep copy (only works if no methods/functions inside)
let deepCopy = JSON.parse(JSON.stringify(objRef1));

objRef1.car.model = "Toyota Innova";  // Does NOT affect deepCopy
